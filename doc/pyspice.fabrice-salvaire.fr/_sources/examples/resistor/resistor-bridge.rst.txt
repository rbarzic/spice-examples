.. code-block:: py3

    
    circuit = Circuit('Resistor Bridge')
    
    circuit.V('input', 1, circuit.gnd, 10@u_V)
    circuit.R(1, 1, 2, 2@u_kÎ©)
    circuit.R(2, 1, 3, 1@u_kÎ©)
    circuit.R(3, 2, circuit.gnd, 1@u_kÎ©)
    circuit.R(4, 3, circuit.gnd, 2@u_kÎ©)
    circuit.R(5, 3, 2, 2@u_kÎ©)
    
    simulator = circuit.simulator(temperature=25, nominal_temperature=25)
    analysis = simulator.operating_point()
    
    for node in analysis.nodes.values():
        print('Node {}: {:4.1f} V'.format(str(node), float(node))) # Fixme: format value + unit
